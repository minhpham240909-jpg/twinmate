**Usage:**
```typescript
// In your service
export class UserService extends DatabaseService {
  async getUser(id: string) {
    return await this.withRetry(async () => {
      const user = await this.db.user.findUnique({ where: { id } })
      if (!user) throw new NotFoundError('User', id)
      return user
    })
  }
}
```
